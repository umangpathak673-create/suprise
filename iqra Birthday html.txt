<script>
let currentStep = 1;

function nextStep(step) {
    let prev = document.getElementById("step" + currentStep);
    let next = document.getElementById("step" + step);

    // Progress bar update
    document.getElementById("progressBar").style.width = ((step - 1) * 25) + "%";

    gsap.to(prev, {opacity:0, y:30, scale:0.95, duration:0.5, onComplete: () => {
        prev.classList.add("hidden");
        
        next.classList.remove("hidden");
        gsap.to(next, {opacity:1, y:0, scale:1, duration:0.6});
    }});

    currentStep = step;
}

/* Polaroid Parallax */
const polaroid = document.getElementById("polaroid");
polaroid?.addEventListener("mousemove", (e) => {
    const rect = polaroid.getBoundingClientRect();
    const x = (e.clientX - rect.left - rect.width/2) / 20;
    const y = (e.clientY - rect.top - rect.height/2) / 20;
    polaroid.style.transform = `rotateX(${-y}deg) rotateY(${x}deg)`;
});
polaroid?.addEventListener("mouseleave", () => {
    polaroid.style.transform = "rotateX(0) rotateY(0)";
});

/* Celebrate Finale */
function celebrate() {
    let btn = document.getElementById("celebrateBtn");
    btn.style.pointerEvents = "none";
    gsap.to(btn, {opacity:0, y:20, duration:0.5});

    gsap.to("#finalMessage", {opacity:1, y:-10, duration:1, delay:0.4});

    // Big Confetti Burst
    for (let i = 0; i < 5; i++) {
        setTimeout(() => {
            confetti({
                particleCount: 150,
                startVelocity: 60,
                spread: 120,
                origin: { y: 1, x: Math.random() < 0.5 ? 0 : 1 }
            });
        }, i * 300);
    }

    // Gentle confetti rain
    const end = Date.now() + 5000;
    (function frame() {
        confetti({
            particleCount: 6,
            spread: 80,
            origin: { x: Math.random(), y: 0 }
        });
        if (Date.now() < end) requestAnimationFrame(frame);
    })();

    // Final heart swirl (Three.js hearts fade away)
    heartsFadeOut();
}

/* ---------------- Three.js Hearts ---------------- */
let scene, camera, renderer, hearts = [];

function createHeartGeometry() {
    const x = 0, y = 0;
    const shape = new THREE.Shape();
    shape.moveTo(x + 5, y + 5);
    shape.bezierCurveTo(x + 5, y + 5, x + 4, y, x, y);
    shape.bezierCurveTo(x - 6, y, x - 6, y + 7, x - 6, y + 7);
    shape.bezierCurveTo(x - 6, y + 11, x - 3, y + 15.4, x + 5, y + 19);
    shape.bezierCurveTo(x + 12, y + 15.4, x + 16, y + 11, x + 16, y + 7);
    shape.bezierCurveTo(x + 16, y + 7, x + 16, y, x + 10, y);
    shape.bezierCurveTo(x + 7, y, x + 5, y + 5, x + 5, y + 5);

    const geo = new THREE.ExtrudeGeometry(shape, {depth: 2, bevelEnabled:false});
    return geo;
}

function initThree() {
    const canvas = document.getElementById("bgCanvas");

    scene = new THREE.Scene();
    camera = new THREE.PerspectiveCamera(75, window.innerWidth/window.innerHeight, 1, 5000);
    camera.position.z = 150;

    renderer = new THREE.WebGLRenderer({canvas, alpha:true});
    renderer.setSize(window.innerWidth, window.innerHeight);

    const heartGeo = createHeartGeometry();
    for (let i = 0; i < 25; i++) {
        const mat = new THREE.MeshStandardMaterial({color: 0xff3e73});
        const h = new THREE.Mesh(heartGeo, mat);
        h.position.set((Math.random()-0.5)*300, (Math.random()-0.5)*200, (Math.random()-0.5)*200);
        h.rotation.set(Math.random()*Math.PI, Math.random()*Math.PI, Math.random()*Math.PI);
        h.scale.setScalar(Math.random()*0.4 + 0.2);
        scene.add(h);
        hearts.push(h);
    }

    const light = new THREE.PointLight(0xffffff, 1.2);
    light.position.set(50,50,50);
    scene.add(light);

    animate();
}

function animate() {
    requestAnimationFrame(animate);
    hearts.forEach((h, i) => {
        h.position.y += Math.sin(Date.now()*0.001 + i)/80;
        h.rotation.y += 0.003;
    });
    renderer.render(scene, camera);
}

function heartsFadeOut() {
    gsap.to(hearts.map(h => h.position), {
        y: "+=200",
        duration: 3,
        ease: "power2.out"
    });
    gsap.to(hearts.map(h => h), {
        opacity: 0,
        duration: 2.5
    });
}

initThree();
